<script>
    // ---------- 衣物数据库 ----------
    // 图标分配：
    // 连衣裙 dress -> 🥻 (传统裙装)
    // 短裙 skirt -> 👗
    // 毛衣 sweater -> 🧶
    // 外套 coat -> 🥼
    // 夹克 jacket -> 🧥
    // 靴子 boot -> 👢 (单数形式)
    // 其他: scarf🧣, hat🧢, T-shirt👕, shirt👔, pants👖, shorts🩳, socks🧦
    // 复数衣物: pants, shorts, socks 提问 "What are these?"
    const ALL_CLOTHES = [
        { id: "jacket", en: "jacket", icon: "🧥", plural: false },
        { id: "scarf", en: "scarf", icon: "🧣", plural: false },
        { id: "hat", en: "hat", icon: "🧢", plural: false },
        { id: "tshirt", en: "T-shirt", icon: "👕", plural: false },
        { id: "shirt", en: "shirt", icon: "👔", plural: false },
        { id: "sweater", en: "sweater", icon: "🧶", plural: false },
        { id: "coat", en: "coat", icon: "🥼", plural: false },
        { id: "skirt", en: "skirt", icon: "👗", plural: false },          // 短裙 👗
        { id: "dress", en: "dress", icon: "🥻", plural: false },          // 连衣裙 🥻
        { id: "pants", en: "pants", icon: "👖", plural: true },
        { id: "shorts", en: "shorts", icon: "🩳", plural: true },
        { id: "socks", en: "socks", icon: "🧦", plural: true },
        { id: "boot", en: "boot", icon: "👢", plural: false }             // 靴子单数
    ];
    const TOTAL_ITEMS = ALL_CLOTHES.length;
    document.getElementById('totalCount').innerText = TOTAL_ITEMS;

    // 游戏状态
    let remainingItems = [];
    let currentCloth = null;
    let currentScore = 0;
    let isGameActive = false;
    let isWashing = false;
    let isWaitingAnswer = false;
    let washTimer = null;

    // DOM 元素
    const drumEl = document.getElementById('drum');
    const clothIconEl = document.getElementById('clothIcon');
    const questionTextEl = document.getElementById('questionText');
    const optionsContainer = document.getElementById('optionsContainer');
    const startBtn = document.getElementById('startBtn');
    const playAgainBtn = document.getElementById('playAgainBtn');
    const scoreSpan = document.getElementById('scoreValue');
    const remainingCountSpan = document.getElementById('remainingCount');
    const feedbackMsg = document.getElementById('feedbackMsg');

    function updateStatsUI() {
        scoreSpan.innerText = currentScore;
        remainingCountSpan.innerText = remainingItems.length;
    }

    function clearOptions() {
        optionsContainer.innerHTML = '';
    }

    function setOptionsEnabled(enabled) {
        const btns = document.querySelectorAll('.option-btn');
        btns.forEach(btn => {
            if (enabled) {
                btn.removeAttribute('disabled');
                btn.classList.remove('disabled-btn');
            } else {
                btn.setAttribute('disabled', 'disabled');
                btn.classList.add('disabled-btn');
            }
        });
    }

    function showRewardFeedback(correctWord) {
        feedbackMsg.innerHTML = `🎉 Great! "${correctWord}" is correct! +1 🎉`;
        feedbackMsg.style.backgroundColor = "#d4ffb0";
        setTimeout(() => {
            if (isGameActive || !isGameActive) {
                feedbackMsg.style.backgroundColor = "#fceb98";
                if (remainingItems.length === 0 && currentScore === TOTAL_ITEMS) {
                    // keep end message
                } else {
                    if(!isWashing) feedbackMsg.innerHTML = "✨ Next piece of clothing! ✨";
                }
            } else {
                feedbackMsg.style.backgroundColor = "#fceb98";
            }
        }, 1200);
        drumEl.classList.add('reward-flash');
        setTimeout(() => drumEl.classList.remove('reward-flash'), 300);
    }

    function showWrongFeedback() {
        feedbackMsg.innerHTML = "❌ Oops! Try again! Look carefully at the clothing! ❌";
        feedbackMsg.style.backgroundColor = "#ffc2a1";
        setTimeout(() => {
            if(isGameActive && !isWashing) feedbackMsg.style.backgroundColor = "#fceb98";
            else if(!isGameActive) feedbackMsg.style.backgroundColor = "#fceb98";
        }, 800);
    }

    function endGameAndShowAgain() {
        isGameActive = false;
        isWaitingAnswer = false;
        if (washTimer) clearTimeout(washTimer);
        setOptionsEnabled(false);
        questionTextEl.innerHTML = "🏆 You washed all clothes! 🏆";
        feedbackMsg.innerHTML = "🎉 Amazing! You learned all the words! Click PLAY AGAIN to challenge again 🎉";
        feedbackMsg.style.backgroundColor = "#ffe3a3";
        playAgainBtn.classList.remove('hidden');
        startBtn.classList.add('hidden');
        clothIconEl.innerHTML = "🏆✨";
    }

    function getQuestionText(cloth) {
        return cloth.plural ? "🤔 What are these?" : "🤔 What is this?";
    }

    // 洗衣循环：1秒洗衣，显示纯图标，无文字
    function startWashCycle() {
        if (washTimer) clearTimeout(washTimer);
        if (!isGameActive) return;
        if (isWashing) return;
        if (remainingItems.length === 0) {
            endGameAndShowAgain();
            return;
        }
        const randomIndex = Math.floor(Math.random() * remainingItems.length);
        currentCloth = { ...remainingItems[randomIndex] };
        
        isWashing = true;
        isWaitingAnswer = false;
        setOptionsEnabled(false);
        clearOptions();
        drumEl.classList.add('washing');
        drumEl.classList.add('drum-washing');
        clothIconEl.innerHTML = "🌀💧🌀";
        questionTextEl.innerHTML = "🌀 Washing machine is spinning... 🌀";
        feedbackMsg.innerHTML = "🧼 Washing the clothes, please wait... 🧼";
        
        washTimer = setTimeout(() => {
            drumEl.classList.remove('washing');
            drumEl.classList.remove('drum-washing');
            if (currentCloth) {
                clothIconEl.innerHTML = currentCloth.icon;
                questionTextEl.innerHTML = getQuestionText(currentCloth);
                feedbackMsg.innerHTML = "👚 Clean clothes! Which word is it? 👕";
                generateOptions(currentCloth.en);
                setOptionsEnabled(true);
                isWashing = false;
                isWaitingAnswer = true;
            } else {
                if(remainingItems.length === 0) endGameAndShowAgain();
                else {
                    isWashing = false;
                    startWashCycle();
                }
            }
        }, 1000);
    }

    // 生成选项（纯文本，无图标）
    function generateOptions(correctEn) {
        clearOptions();
        let allEnWords = ALL_CLOTHES.map(c => c.en);
        let candidates = allEnWords.filter(word => word !== correctEn);
        let shuffledCandidates = [...candidates];
        for (let i = shuffledCandidates.length - 1; i > 0; i--) {
            const j = Math.floor(Math.random() * (i + 1));
            [shuffledCandidates[i], shuffledCandidates[j]] = [shuffledCandidates[j], shuffledCandidates[i]];
        }
        let wrongOptions = shuffledCandidates.slice(0, 3);
        let options = [correctEn, ...wrongOptions];
        for (let i = options.length - 1; i > 0; i--) {
            const j = Math.floor(Math.random() * (i + 1));
            [options[i], options[j]] = [options[j], options[i]];
        }
        
        options.forEach(optionWord => {
            const btn = document.createElement('button');
            btn.className = 'option-btn';
            btn.textContent = optionWord;
            btn.addEventListener('click', (e) => {
                e.stopPropagation();
                if (!isGameActive || isWashing || !isWaitingAnswer) {
                    feedbackMsg.innerHTML = "⏳ Please wait until washing is finished!";
                    return;
                }
                if (optionWord === correctEn) {
                    if (currentCloth && remainingItems.length > 0) {
                        const indexInRemaining = remainingItems.findIndex(item => item.id === currentCloth.id);
                        if (indexInRemaining !== -1) {
                            remainingItems.splice(indexInRemaining, 1);
                            currentScore++;
                            updateStatsUI();
                            showRewardFeedback(correctEn);
                            isWaitingAnswer = false;
                            setOptionsEnabled(false);
                            clearOptions();
                            if (remainingItems.length === 0 && currentScore === TOTAL_ITEMS) {
                                endGameAndShowAgain();
                            } else {
                                startWashCycle();
                            }
                        } else {
                            isWaitingAnswer = false;
                            startWashCycle();
                        }
                    } else {
                        if(remainingItems.length === 0) endGameAndShowAgain();
                        else startWashCycle();
                    }
                } else {
                    showWrongFeedback();
                    drumEl.style.transform = "rotate(2deg)";
                    setTimeout(() => { if(drumEl) drumEl.style.transform = ""; }, 150);
                }
            });
            optionsContainer.appendChild(btn);
        });
    }

    function resetAndStart() {
        if (washTimer) clearTimeout(washTimer);
        isGameActive = false;
        isWashing = false;
        isWaitingAnswer = false;
        remainingItems = ALL_CLOTHES.map(item => ({ ...item }));
        currentScore = 0;
        currentCloth = null;
        updateStatsUI();
        clearOptions();
        setOptionsEnabled(false);
        drumEl.classList.remove('washing', 'drum-washing', 'reward-flash');
        clothIconEl.innerHTML = "🧺";
        questionTextEl.innerHTML = "🤔 What is this?";
        feedbackMsg.innerHTML = "🎵 Press START, let's wash clothes and learn words! 🎵";
        feedbackMsg.style.backgroundColor = "#fceb98";
        playAgainBtn.classList.add('hidden');
        startBtn.classList.remove('hidden');
        isGameActive = false;
    }

    function launchGame() {
        if (isGameActive) {
            feedbackMsg.innerHTML = "Game already started! Answer the question!";
            return;
        }
        resetAndStart();
        remainingItems = ALL_CLOTHES.map(item => ({ ...item }));
        currentScore = 0;
        updateStatsUI();
        isGameActive = true;
        isWashing = false;
        isWaitingAnswer = false;
        if (washTimer) clearTimeout(washTimer);
        clearOptions();
        setOptionsEnabled(false);
        feedbackMsg.innerHTML = "🚀 Washer starts! Let's go! 🚀";
        startBtn.classList.add('hidden');
        playAgainBtn.classList.add('hidden');
        startWashCycle();
    }

    function replayGame() {
        resetAndStart();
        isGameActive = false;
        startBtn.classList.remove('hidden');
        playAgainBtn.classList.add('hidden');
        feedbackMsg.innerHTML = "🎈 New round! Click START to wash and learn! 🎈";
        clothIconEl.innerHTML = "🧺";
        questionTextEl.innerHTML = "🤔 What is this?";
        clearOptions();
        setOptionsEnabled(false);
        if (washTimer) clearTimeout(washTimer);
    }

    startBtn.addEventListener('click', launchGame);
    playAgainBtn.addEventListener('click', replayGame);

    resetAndStart();
    setOptionsEnabled(false);
</script>
